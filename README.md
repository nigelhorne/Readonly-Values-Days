# NAME

Readonly::Values::Days - Days Constants

# VERSION

Version 0.01

# SYNOPSIS

    use Readonly::Values::Days;

    # Iterate full day names
    for my $name (@day_names) {
        printf "%-9s => %2d\n", ucfirst($name), $days{$name};
    }
